import pandas as pd

def filter_by_datetime(df: pd.DataFrame, datetime_col: str, threshold: str):
    """
    Filter rows in a DataFrame where the datetime column is less than the given threshold.

    Args:
        df (pd.DataFrame): Input DataFrame containing a datetime column.
        datetime_col (str): Name of the datetime column in the DataFrame.
        threshold (str): The threshold datetime in 'YYYY-MM-DD HH:MM:SS' format.

    Returns:
        pd.DataFrame: Filtered DataFrame with rows where datetime_col < threshold.
    """
    # Convert the datetime column to pandas datetime format
    df[datetime_col] = pd.to_datetime(df[datetime_col])
    
    # Convert the threshold to a datetime object
    threshold_datetime = pd.to_datetime(threshold)
    
    # Filter rows where datetime_col is less than the threshold
    filtered_df = df[df[datetime_col] < threshold_datetime]
    
    return filtered_df
